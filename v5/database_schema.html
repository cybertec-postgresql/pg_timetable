9. Database Schema

pg_timetable is a database driven application. During the first start the necessary schema is created if absent.

9.1. Main tables and objects

136CREATE OR REPLACE FUNCTION timetable.try_lock_client_name(worker_pid BIGINT, worker_name TEXT)
137RETURNS bool AS
138$CODE$
139BEGIN
140    IF pg_is_in_recovery() THEN
141        RAISE NOTICE 'Cannot obtain lock on a replica. Please, use the primary node';
142        RETURN FALSE;
143    END IF;
144    -- remove disconnected sessions
145    DELETE
146        FROM timetable.active_session
147        WHERE server_pid NOT IN (
148            SELECT pid
149            FROM pg_catalog.pg_stat_activity
150            WHERE application_name = 'pg_timetable'
151        );
152    DELETE 
153        FROM timetable.active_chain 
154        WHERE client_name NOT IN (
155            SELECT client_name FROM timetable.active_session
156        );
157    -- check if there any active sessions with the client name but different client pid
158    PERFORM 1
159        FROM timetable.active_session s
160        WHERE
161            s.client_pid <> worker_pid
162            AND s.client_name = worker_name
163        LIMIT 1;
164    IF FOUND THEN
165        RAISE NOTICE 'Another client is already connected to server with name: %', worker_name;
166        RETURN FALSE;
167    END IF;
168    -- insert current session information
169    INSERT INTO timetable.active_session(client_pid, client_name, server_pid) VALUES (worker_pid, worker_name, pg_backend_pid());
170    RETURN TRUE;
171END;
172$CODE$
173STRICT
174LANGUAGE plpgsql;

9.2. Jobs related functions

  1-- add_task() will add a task to the same chain as the task with `parent_id`
  2CREATE OR REPLACE FUNCTION timetable.add_task(
  3    IN kind timetable.command_kind,
  4    IN command TEXT, 
  5    IN parent_id BIGINT,
  6    IN order_delta DOUBLE PRECISION DEFAULT 10
  7) RETURNS BIGINT AS $$
  8    INSERT INTO timetable.task (chain_id, task_order, kind, command) 
  9	SELECT chain_id, task_order + $4, $1, $2 FROM timetable.task WHERE task_id = $3
 10	RETURNING task_id
 11$$ LANGUAGE SQL;

 15-- add_job() will add one-task chain to the system
 16CREATE OR REPLACE FUNCTION timetable.add_job(
 17    job_name            TEXT,
 18    job_schedule        timetable.cron,
 19    job_command         TEXT,
 20    job_parameters      JSONB DEFAULT NULL,
 21    job_kind            timetable.command_kind DEFAULT 'SQL'::timetable.command_kind,
 22    job_client_name     TEXT DEFAULT NULL,
 23    job_max_instances   INTEGER DEFAULT NULL,
 24    job_live            BOOLEAN DEFAULT TRUE,
 25    job_self_destruct   BOOLEAN DEFAULT FALSE,
 26    job_ignore_errors   BOOLEAN DEFAULT TRUE,
 27    job_exclusive       BOOLEAN DEFAULT FALSE,
 28    job_on_error        TEXT DEFAULT NULL
 29) RETURNS BIGINT AS $$
 30    WITH 
 31        cte_chain (v_chain_id) AS (
 32            INSERT INTO timetable.chain (chain_name, run_at, max_instances, live, self_destruct, client_name, exclusive_execution, on_error) 
 33            VALUES (job_name, job_schedule,job_max_instances, job_live, job_self_destruct, job_client_name, job_exclusive, job_on_error)
 34            RETURNING chain_id
 35        ),
 36        cte_task(v_task_id) AS (
 37            INSERT INTO timetable.task (chain_id, task_order, kind, command, ignore_error, autonomous)
 38            SELECT v_chain_id, 10, job_kind, job_command, job_ignore_errors, TRUE
 39            FROM cte_chain
 40            RETURNING task_id
 41        ),
 42        cte_param AS (
 43            INSERT INTO timetable.parameter (task_id, order_id, value)
 44            SELECT v_task_id, 1, job_parameters FROM cte_task, cte_chain
 45        )
 46        SELECT v_chain_id FROM cte_chain
 47$$ LANGUAGE SQL;
 48
 49COMMENT ON FUNCTION timetable.add_job IS 'Add one-task chain (aka job) to the system';
 50
 51-- notify_chain_start() will send notification to the worker to start the chain
 52CREATE OR REPLACE FUNCTION timetable.notify_chain_start(
 53    chain_id    BIGINT, 
 54    worker_name TEXT,
 55    start_delay INTERVAL DEFAULT NULL
 56) RETURNS void AS $$
 57    SELECT pg_notify(
 58        worker_name, 
 59        format('{"ConfigID": %s, "Command": "START", "Ts": %s, "Delay": %s}', 
 60            chain_id, 
 61            EXTRACT(epoch FROM clock_timestamp())::bigint,
 62            COALESCE(EXTRACT(epoch FROM start_delay)::bigint, 0)
 63        )
 64    )
 65$$ LANGUAGE SQL;

 84-- move_task_up() will switch the order of the task execution with a previous task within the chain
 85CREATE OR REPLACE FUNCTION timetable.move_task_up(IN task_id BIGINT) RETURNS boolean AS $$
 86	WITH current_task (ct_chain_id, ct_id, ct_order) AS (
 87		SELECT chain_id, task_id, task_order FROM timetable.task WHERE task_id = $1
 88	),
 89	tasks(t_id, t_new_order) AS (
 90		SELECT task_id, COALESCE(LAG(task_order) OVER w, LEAD(task_order) OVER w)
 91		FROM timetable.task t, current_task ct
 92		WHERE chain_id = ct_chain_id AND (task_order < ct_order OR task_id = ct_id)
 93		WINDOW w AS (PARTITION BY chain_id ORDER BY ABS(task_order - ct_order))
 94		LIMIT 2
 95	),
 96	upd AS (
 97		UPDATE timetable.task t SET task_order = t_new_order
 98		FROM tasks WHERE tasks.t_id = t.task_id AND tasks.t_new_order IS NOT NULL
 99		RETURNING true
100	)
101	SELECT COUNT(*) > 0 FROM upd
102$$ LANGUAGE SQL;
103
104COMMENT ON FUNCTION timetable.move_task_up IS 'Switch the order of the task execution with a previous task within the chain';
105
106-- move_task_down() will switch the order of the task execution with a following task within the chain
107CREATE OR REPLACE FUNCTION timetable.move_task_down(IN task_id BIGINT) RETURNS boolean AS $$
108	WITH current_task (ct_chain_id, ct_id, ct_order) AS (
109		SELECT chain_id, task_id, task_order FROM timetable.task WHERE task_id = $1
110	),
111	tasks(t_id, t_new_order) AS (
112		SELECT task_id, COALESCE(LAG(task_order) OVER w, LEAD(task_order) OVER w)
113		FROM timetable.task t, current_task ct
114		WHERE chain_id = ct_chain_id AND (task_order > ct_order OR task_id = ct_id)
115		WINDOW w AS (PARTITION BY chain_id ORDER BY ABS(task_order - ct_order))
116		LIMIT 2
117	),
118	upd AS (
119		UPDATE timetable.task t SET task_order = t_new_order
120		FROM tasks WHERE tasks.t_id = t.task_id AND tasks.t_new_order IS NOT NULL
121		RETURNING true
122	)
123	SELECT COUNT(*) > 0 FROM upd
124$$ LANGUAGE SQL;

128-- delete_job() will delete the chain and its tasks from the system
129CREATE OR REPLACE FUNCTION timetable.delete_job(IN job_name TEXT) RETURNS boolean AS $$
130    WITH del_chain AS (DELETE FROM timetable.chain WHERE chain.chain_name = $1 RETURNING chain_id)
131    SELECT EXISTS(SELECT 1 FROM del_chain)
132$$ LANGUAGE SQL;

134COMMENT ON FUNCTION timetable.delete_job IS 'Delete the chain and its tasks from the system';

136-- delete_task() will delete the task from a chain

139    SELECT EXISTS(SELECT 1 FROM del_task)
