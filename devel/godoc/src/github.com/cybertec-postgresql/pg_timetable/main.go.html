    27  // SetupCloseHandler creates a 'listener' on a new goroutine which will notify the
    28  // program if it receives an interrupt from the OS. We then handle this by calling
    29  // our clean up procedure and exiting the program.
    30  func SetupCloseHandler(cancel context.CancelFunc) {
    31  	c := make(chan os.Signal, 2)
    32  	signal.Notify(c, os.Interrupt, syscall.SIGTERM)
    33  	go func() {
    34  		<-c
    35  		cancel()
    36  		exitCode = ExitCodeUserCancel
    37  	}()
    38  }
