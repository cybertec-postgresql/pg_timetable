    73  	if err := pge.SetRole(ctx, tx, task.RunAs); err != nil {
    74  		return "", err
    75  	}
    76  	if task.IgnoreError {
    77  		pge.MustSavepoint(ctx, tx, task.TaskID)
    78  	}
    79  	pge.SetCurrentTaskContext(ctx, tx, task.ChainID, task.TaskID)
    80  	out, err = pge.ExecuteSQLCommand(ctx, tx, task.Script, paramValues)
    81  	if err != nil && task.IgnoreError {
    82  		pge.MustRollbackToSavepoint(ctx, tx, task.TaskID)

    86  	}

   120  // ExecuteSQLCommand executes chain command with parameters inside transaction
   121  func (pge *PgEngine) ExecuteSQLCommand(ctx context.Context, executor executor, command string, paramValues []string) (out string, err error) {
   122  	var ct pgconn.CommandTag
   123  	var params []interface{}
   124  	if strings.TrimSpace(command) == "" {
   125  		return "", errors.New("SQL command cannot be empty")
   126  	}
   127  	if len(paramValues) == 0 { //mimic empty param
   128  		ct, err = executor.Exec(ctx, command)
   129  		out = ct.String()
   130  		return
   131  	}
   132  	for _, val := range paramValues {
   133  		if val > "" {
   134  			if err = json.Unmarshal([]byte(val), &params); err != nil {
   135  				return
   136  			}
   137  			ct, err = executor.Exec(ctx, command, params...)
   138  			out = out + ct.String() + "\n"
   139  		}
   140  	}
   141  	return
   142  }
