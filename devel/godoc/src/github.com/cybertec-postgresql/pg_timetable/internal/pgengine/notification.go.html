    16  // ChainSignal used to hold asynchronous notifications from PostgreSQL server
    17  type ChainSignal struct {
    18  	ConfigID int    // chain configuration ifentifier
    19  	Command  string // allowed: START, STOP
    20  	Ts       int64  // timestamp NOTIFY sent
    21  	Delay    int64  // delay in seconds before start

    24  // Since there are usually multiple opened connections to the database, all of them will receive NOTIFY messages.
    25  // To process each NOTIFY message only once we store each message with TTL 1 minute because the max idle period for a
    26  // a connection is the main loop period of 1 minute.
    27  var mutex sync.Mutex
    28  var notifications = func() (m map[ChainSignal]struct{}) {
    29  	m = make(map[ChainSignal]struct{})
    30  	go func() {
    31  		for now := range time.Tick(time.Duration(NotifyTTL) * time.Second) {
    32  			mutex.Lock()
    33  			for k := range m {
    34  				if now.Unix()-k.Ts > NotifyTTL {
    35  					delete(m, k)
    36  				}
    37  			}
    38  			mutex.Unlock()
    39  		}
    40  	}()
    41  	return
    42  }()

    44  // NotificationHandler consumes notifications from the PostgreSQL server
    45  func (pge *PgEngine) NotificationHandler(c *pgconn.PgConn, n *pgconn.Notification) {
    46  	l := pge.l.WithField("pid", c.PID()).WithField("notification", *n)
    47  	l.Debug("Notification received")
    48  	var signal ChainSignal
    49  	var err error
    50  	if err = json.Unmarshal([]byte(n.Payload), &signal); err == nil {
    51  		mutex.Lock()
    52  		if _, ok := notifications[signal]; ok {
    53  			l.WithField("handled", notifications).Debug("Notification already handled")
    54  			mutex.Unlock()
    55  			return
    56  		}
    57  		notifications[signal] = struct{}{}
    58  		mutex.Unlock()
    59  		switch signal.Command {
    60  		case "STOP", "START":
    61  			if signal.ConfigID > 0 {
    62  				l.WithField("signal", signal).Info("Adding asynchronous chain to working queue")
    63  				pge.chainSignalChan <- signal
    64  				return
    65  			}
    66  			err = fmt.Errorf("unknown chain ID: %d", signal.ConfigID)
    67  		default:
    68  			err = fmt.Errorf("unknown command: %s", signal.Command)
    69  		}
    70  	}
    71  	l.WithError(err).Error("syntax error in payload")
    72  }
