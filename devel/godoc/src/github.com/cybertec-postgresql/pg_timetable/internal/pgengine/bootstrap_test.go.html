    95  func (r *mockpgrow) Scan(dest ...any) error {
    96  	if len(r.results) > 0 {
    97  		if err, ok := r.results[0].(error); ok {
    98  			r.results = r.results[1:]
    99  			return err
   100  		}
   101  		destv := reflect.ValueOf(dest[0])
   102  		typ := destv.Type()
   103  		if typ.Kind() != reflect.Ptr {
   104  			return fmt.Errorf("dest must be pointer; got %T", destv)
   105  		}
   106  		destv.Elem().Set(reflect.ValueOf(r.results[0]))
   107  		r.results = r.results[1:]
   108  		return nil
   109  	}
   110  	return errors.New("mockpgrow error")
   111  }
