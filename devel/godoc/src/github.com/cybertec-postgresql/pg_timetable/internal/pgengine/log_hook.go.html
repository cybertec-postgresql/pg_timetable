    45  // Fire adds logrus log message to the internal queue for processing
    46  func (hook *LogHook) Fire(entry *logrus.Entry) error {
    47  	if hook.ctx.Err() != nil {
    48  		return nil
    49  	}
    50  	select {
    51  	case hook.input <- *entry:
    52  		// entry sent
    53  	case <-time.After(hook.highLoadTimeout):
    54  		// entry dropped due to a huge load, check stdout or file for detailed log
    55  	}
    56  	select {
    57  	case err := <-hook.lastError:
    58  		return err
    59  	default:
    60  		return nil
    61  	}
    62  }

    88  // poll checks for incoming messages and caches them internally
    89  // until either a maximum amount is reached, or a timeout occurs.
    90  func (hook *LogHook) poll(input <-chan logrus.Entry) {
    91  	cache := make([]logrus.Entry, 0, hook.cacheLimit)
    92  	tick := time.NewTicker(hook.cacheTimeout)
    93  
    94  	for {
    95  		select {
    96  		case <-hook.ctx.Done(): //check context with high priority
    97  			return
    98  		default:
    99  			select {
   100  			case entry := <-input:
   101  				cache = append(cache, entry)
   102  				if len(cache) < hook.cacheLimit {
   103  					break
   104  				}
   105  				tick.Stop()
   106  				hook.send(cache)
   107  				cache = cache[:0]
   108  				tick = time.NewTicker(hook.cacheTimeout)
   109  			case <-tick.C:
   110  				hook.send(cache)
   111  				cache = cache[:0]
   112  			case <-hook.ctx.Done():
   113  				return
   114  			}
   115  		}
   116  	}
   117  }

   133  // send sends cached messages to the postgres server
   134  func (hook *LogHook) send(cache []logrus.Entry) {
   135  	if len(cache) == 0 {
   136  		return // Nothing to do here.
   137  	}
   138  	_, err := hook.db.CopyFrom(
   139  		hook.ctx,
   140  		pgx.Identifier{"timetable", "log"},
   141  		[]string{"ts", "client_name", "pid", "log_level", "message", "message_data"},
   142  		pgx.CopyFromSlice(len(cache),
   143  			func(i int) ([]any, error) {
   144  				jsonData, err := json.Marshal(cache[i].Data)
   145  				if err != nil {
   146  					return nil, err
   147  				}
   148  				return []any{cache[i].Time,
   149  					hook.client,
   150  					hook.pid,
   151  					adaptEntryLevel(cache[i].Level),
   152  					cache[i].Message,
   153  					jsonData}, nil
   154  			}),
   155  	)
   156  	if err != nil {
   157  		select {
   158  		case hook.lastError <- err:
   159  			//error sent to the logger
   160  		default:
   161  			//there is unprocessed error already
   162  		}
   163  	}
   164  }
