    36  	// Parse YAML file
    37  	yamlConfig, err := ParseYamlFile(filePath)

    40  	}
    41  
    42  	// Import chains
    43  	for _, yamlChain := range yamlConfig.Chains {
    44  		// Delete existing chain if replace mode
    45  		if replace {
    46  			_, _ = pge.ConfigDb.Exec(ctx, "SELECT timetable.delete_job($1)", yamlChain.ChainName)
    47  		}
    48  
    49  		// Check if chain exists
    50  		var exists bool
    51  		err := pge.ConfigDb.QueryRow(ctx,
    52  			"SELECT EXISTS(SELECT 1 FROM timetable.chain WHERE chain_name = $1)",
    53  			yamlChain.ChainName).Scan(&exists)
    54  		if err != nil {
    55  			return fmt.Errorf("failed to check if chain exists: %w", err)
    56  		}
    57  		if exists && !replace {
    58  			return fmt.Errorf("chain '%s' already exists (use --replace flag to overwrite)", yamlChain.ChainName)
    59  		}
    60  
    61  		// Multi-task chain - use direct SQL
    62  		chainID, err := pge.createChainFromYaml(ctx, &yamlChain)
    63  		if err != nil {
    64  			return fmt.Errorf("failed to create multi-task chain %s: %w", yamlChain.ChainName, err)
    65  		}
    66  		pge.l.WithField("chain", yamlChain.ChainName).WithField("chain_id", chainID).Info("Created multi-task chain")
    67  	}
    68  
    69  	pge.l.WithField("chains", len(yamlConfig.Chains)).WithField("file", filePath).Info("Successfully imported YAML chains")
    70  	return nil

    74  func (pge *PgEngine) createChainFromYaml(ctx context.Context, yamlChain *YamlChain) (int64, error) {
    75  	// Insert chain
    76  	var chainID int64
    77  	err := pge.ConfigDb.QueryRow(ctx, `
    78  		INSERT INTO timetable.chain (
    79  			chain_name, run_at, max_instances, timeout, live, 
    80  			self_destruct, exclusive_execution, client_name, on_error
    81  		) VALUES ($1, $2, $3, $4, $5, $6, $7, $8, $9) 
    82  		RETURNING chain_id`,
    83  		yamlChain.ChainName,
    84  		yamlChain.Schedule,
    85  		yamlChain.MaxInstances,
    86  		yamlChain.Timeout,
    87  		yamlChain.Live,
    88  		yamlChain.SelfDestruct,
    89  		yamlChain.ExclusiveExecution,
    90  		nullString(yamlChain.ClientName),
    91  		nullString(yamlChain.OnError)).Scan(&chainID)
    92  	if err != nil {
    93  		return 0, fmt.Errorf("failed to insert chain: %w", err)
    94  	}
    95  
    96  	// Insert tasks
    97  	for i, task := range yamlChain.Tasks {
    98  		taskOrder := float64((i + 1) * 10)
    99  
   100  		var taskID int64
   101  		err := pge.ConfigDb.QueryRow(ctx, `
   102  			INSERT INTO timetable.task (
   103  				chain_id, task_order, task_name, kind, command, 
   104  				run_as, database_connection, ignore_error, autonomous, timeout
   105  			) VALUES ($1, $2, $3, $4::timetable.command_kind, $5, $6, $7, $8, $9, $10) 
   106  			RETURNING task_id`,
   107  			chainID,
   108  			taskOrder,
   109  			nullString(task.TaskName),
   110  			task.Kind,
   111  			task.Command,
   112  			nullString(task.RunAs),
   113  			nullString(task.ConnectString),
   114  			task.IgnoreError,
   115  			task.Autonomous,
   116  			task.Timeout).Scan(&taskID)
   117  		if err != nil {
   118  			return 0, fmt.Errorf("failed to insert task %d: %w", i+1, err)
   119  		}
   120  
   121  		// Insert parameters if any
   122  		if len(task.Parameters) > 0 {
   123  			params, err := task.ToSQLParameters()
   124  			if err != nil {
   125  				return 0, fmt.Errorf("failed to convert parameters: %w", err)
   126  			}
   127  			_, err = pge.ConfigDb.Exec(ctx,
   128  				"INSERT INTO timetable.parameter (task_id, order_id, value) VALUES ($1, 1, $2::jsonb)",
   129  				taskID, params)
   130  			if err != nil {
   131  				return 0, fmt.Errorf("failed to insert parameters: %w", err)
   132  			}
   133  		}
   134  	}
   135  
   136  	return chainID, nil
   137  }

   147  // ValidateChain validates a YAML chain configuration
   148  func (c *YamlChain) ValidateChain() error {
   149  	if c.ChainName == "" {
   150  		return fmt.Errorf("chain name is required")
   151  	}
   152  
   153  	if c.Schedule == "" {
   154  		return fmt.Errorf("chain schedule is required")
   155  	}
   156  
   157  	// Validate cron format
   158  	specialSchedules := []string{"@reboot", "@after", "@every"}
   159  	isSpecial := false
   160  	for _, s := range specialSchedules {
   161  		if strings.HasPrefix(c.Schedule, s) {
   162  			isSpecial = true
   163  			break
   164  		}
   165  	}
   166  
   167  	if !isSpecial {
   168  		fields := strings.Fields(c.Schedule)
   169  		if len(fields) != 5 {
   170  			return fmt.Errorf("invalid cron format: %s (expected 5 fields)", c.Schedule)
   171  		}
   172  	}
   173  
   174  	if len(c.Tasks) == 0 {
   175  		return fmt.Errorf("chain must have at least one task")
   176  	}
   177  
   178  	// Validate each task
   179  	for i, task := range c.Tasks {
   180  		if err := task.ValidateTask(); err != nil {
   181  			return fmt.Errorf("task %d: %w", i+1, err)
   182  		}
   183  	}
   184  
   185  	return nil
   186  }

   226  // ParseYamlFile parses a YAML file and returns the configuration
   227  func ParseYamlFile(filePath string) (*YamlConfig, error) {
   228  	// Check if file exists
   229  	if _, err := os.Stat(filePath); os.IsNotExist(err) {
   230  		return nil, fmt.Errorf("file not found: %s", filePath)
   231  	}
   232  
   233  	// Check file extension
   234  	ext := strings.ToLower(filepath.Ext(filePath))
   235  	if ext != ".yaml" && ext != ".yml" {
   236  		return nil, fmt.Errorf("file must have .yaml or .yml extension: %s", filePath)
   237  	}
   238  
   239  	// Read file
   240  	data, err := os.ReadFile(filePath)
   241  	if err != nil {
   242  		return nil, fmt.Errorf("failed to read file: %w", err)
   243  	}
   244  
   245  	// Parse YAML
   246  	var config YamlConfig
   247  	if err := yaml.Unmarshal(data, &config); err != nil {
   248  		return nil, fmt.Errorf("failed to parse YAML: %w", err)
   249  	}
   250  
   251  	// Set defaults and validate
   252  	for i := range config.Chains {
   253  		chain := &config.Chains[i]
   254  		chain.SetDefaults()
   255  		if err := chain.ValidateChain(); err != nil {
   256  			return nil, fmt.Errorf("chain %d (%s): %w", i+1, chain.ChainName, err)
   257  		}
   258  	}
   259  
   260  	return &config, nil
   261  }

   263  // ToSQLParameters converts YAML parameters to SQL-compatible format
   264  func (t *YamlTask) ToSQLParameters() (string, error) {
   265  	if len(t.Parameters) == 0 {
   266  		return "", nil
   267  	}
   268  
   269  	// Convert to JSON array format for PostgreSQL
   270  	params := make([]string, len(t.Parameters))
   271  	for i, param := range t.Parameters {
   272  		switch v := param.(type) {
   273  		case string:
   274  			params[i] = fmt.Sprintf(`"%s"`, strings.ReplaceAll(v, `"`, `\"`))
   275  		case int, int32, int64:
   276  			params[i] = fmt.Sprintf("%v", v)
   277  		case float32, float64:
   278  			params[i] = fmt.Sprintf("%v", v)
   279  		case bool:
   280  			params[i] = fmt.Sprintf("%t", v)
   281  		default:
   282  			params[i] = fmt.Sprintf(`"%v"`, v)
   283  		}
   284  	}
   285  
   286  	return fmt.Sprintf("[%s]", strings.Join(params, ", ")), nil
   287  }
