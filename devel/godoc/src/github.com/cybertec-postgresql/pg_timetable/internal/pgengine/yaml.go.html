    75  	// Insert chain

    78  			chain_name, run_at, max_instances, timeout, live, 
    79  			self_destruct, exclusive_execution, client_name, on_error
    80  		) VALUES ($1, $2, $3, $4, $5, $6, $7, $8, $9) 
    81  		RETURNING chain_id`,
    82  		yamlChain.ChainName,
    83  		yamlChain.Schedule,
    84  		yamlChain.MaxInstances,
    85  		yamlChain.Timeout,
    86  		yamlChain.Live,
    87  		yamlChain.SelfDestruct,
    88  		yamlChain.ExclusiveExecution,
    89  		nullString(yamlChain.ClientName),
    90  		nullString(yamlChain.OnError)).Scan(&chainID)
    91  	if err != nil {
    92  		return 0, fmt.Errorf("failed to insert chain: %w", err)
    93  	}
    94  
    95  	// Insert tasks
    96  	for i, task := range yamlChain.Tasks {
    97  		taskOrder := float64((i + 1) * 10)
    98  
    99  		var taskID int64
   100  		err := pge.ConfigDb.QueryRow(ctx, `
   101  			INSERT INTO timetable.task (
   102  				chain_id, task_order, task_name, kind, command, 
   103  				run_as, database_connection, ignore_error, autonomous, timeout
   104  			) VALUES ($1, $2, $3, $4::timetable.command_kind, $5, $6, $7, $8, $9, $10) 
   105  			RETURNING task_id`,
   106  			chainID,
   107  			taskOrder,
   108  			nullString(task.TaskName),
   109  			task.Kind,
   110  			task.Command,
   111  			nullString(task.RunAs),
   112  			nullString(task.ConnectString),
   113  			task.IgnoreError,
   114  			task.Autonomous,
   115  			task.Timeout).Scan(&taskID)
   116  		if err != nil {
   117  			return 0, fmt.Errorf("failed to insert task %d: %w", i+1, err)
   118  		}
   119  
   120  		// Insert parameters if any
   121  		if len(task.Parameters) > 0 {
   122  			for paramIndex, param := range task.Parameters {
   123  				orderID := paramIndex + 1
   124  				
   125  				// Convert parameter to JSON for JSONB storage
   126  				jsonValue, err := json.Marshal(param)
   127  				if err != nil {
   128  					return 0, fmt.Errorf("failed to marshal parameter %d to JSON: %w", orderID, err)
   129  				}
   130  				
   131  				_, err = pge.ConfigDb.Exec(ctx,
   132  					"INSERT INTO timetable.parameter (task_id, order_id, value) VALUES ($1, $2, $3::jsonb)",
   133  					taskID, orderID, string(jsonValue))
   134  				if err != nil {
   135  					return 0, fmt.Errorf("failed to insert parameter %d: %w", orderID, err)
   136  				}
   137  			}
   138  		}
   139  	}
   140  
   141  	return chainID, nil
   142  }

   152  // ValidateChain validates a YAML chain configuration
   153  func (c *YamlChain) ValidateChain() error {
   154  	if c.ChainName == "" {
   155  		return fmt.Errorf("chain name is required")
   156  	}
   157  
   158  	if c.Schedule == "" {
   159  		return fmt.Errorf("chain schedule is required")
   160  	}
   161  
   162  	// Validate cron format
   163  	specialSchedules := []string{"@reboot", "@after", "@every"}
   164  	isSpecial := false
   165  	for _, s := range specialSchedules {
   166  		if strings.HasPrefix(c.Schedule, s) {
   167  			isSpecial = true
   168  			break
   169  		}
   170  	}
   171  
   172  	if !isSpecial {
   173  		fields := strings.Fields(c.Schedule)
   174  		if len(fields) != 5 {
   175  			return fmt.Errorf("invalid cron format: %s (expected 5 fields)", c.Schedule)
   176  		}
   177  	}
   178  
   179  	if len(c.Tasks) == 0 {
   180  		return fmt.Errorf("chain must have at least one task")
   181  	}
   182  
   183  	// Validate each task
   184  	for i, task := range c.Tasks {
   185  		if err := task.ValidateTask(); err != nil {
   186  			return fmt.Errorf("task %d: %w", i+1, err)
   187  		}
   188  	}
   189  
   190  	return nil
   191  }

   231  // ParseYamlFile parses a YAML file and returns the configuration
   232  func ParseYamlFile(filePath string) (*YamlConfig, error) {
   233  	// Check if file exists
   234  	if _, err := os.Stat(filePath); os.IsNotExist(err) {
   235  		return nil, fmt.Errorf("file not found: %s", filePath)
   236  	}
   237  
   238  	// Read file
   239  	data, err := os.ReadFile(filePath)
   240  	if err != nil {
   241  		return nil, fmt.Errorf("failed to read file: %w", err)
   242  	}
   243  
   244  	// Parse YAML
   245  	var config YamlConfig
   246  	if err := yaml.Unmarshal(data, &config); err != nil {
   247  		return nil, fmt.Errorf("failed to parse YAML: %w", err)
   248  	}
   249  
   250  	// Set defaults and validate
   251  	for i := range config.Chains {
   252  		chain := &config.Chains[i]
   253  		chain.SetDefaults()
   254  		if err := chain.ValidateChain(); err != nil {
   255  			return nil, fmt.Errorf("chain %d (%s): %w", i+1, chain.ChainName, err)
   256  		}
   257  	}
   258  
   259  	return &config, nil
   260  }
