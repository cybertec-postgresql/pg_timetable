    81  func New(ctx context.Context, cmdOpts config.CmdOptions, logger log.LoggerHookerIface) (*PgEngine, error) {
    82  	var (
    83  		err error
    84  	)
    85  	pge := &PgEngine{
    86  		l:               logger,
    87  		ConfigDb:        nil,
    88  		CmdOptions:      cmdOpts,
    89  		chainSignalChan: make(chan ChainSignal, 64),
    90  	}
    91  	pge.l.WithField("sid", pge.Getsid()).Info("Starting new session... ")
    92  	config := pge.getPgxConnConfig()
    93  	if err = retry.Do(ctx, backoff, func(ctx context.Context) error {
    94  		if pge.ConfigDb, err = pgxpool.NewWithConfig(ctx, config); err == nil {
    95  			err = pge.ConfigDb.Ping(ctx)
    96  		}
    97  		if err != nil {
    98  			pge.l.WithError(err).Error("Connection failed")
    99  			pge.l.Info("Sleeping before reconnecting...")
   100  			return retry.RetryableError(err)
   101  		}
   102  		return nil
   103  	}); err != nil {
   104  		return nil, err
   105  	}
   106  	pge.l.Info("Database connection established")
   107  	if err := pge.ExecuteSchemaScripts(ctx); err != nil {
   108  		return nil, err
   109  	}
   110  	pge.AddLogHook(ctx) //schema exists, we can log now
   111  	if cmdOpts.Start.File != "" {
   112  		if err := pge.ExecuteFileScript(ctx, cmdOpts); err != nil {
   113  			return nil, err
   114  		}
   115  	}
   116  	return pge, nil
   117  }

   135  func (pge *PgEngine) getPgxConnConfig() *pgxpool.Config {
   136  	connConfig, err := pgxpool.ParseConfig(pge.ConnStr)
   137  	if err != nil {
   138  		pge.l.WithError(err).Error("Cannot parse connection string")
   139  		return nil
   140  	}
   141  	// in the worst scenario we need separate connections for each of workers,
   142  	// separate connection for Scheduler.retrieveChainsAndRun(),
   143  	// separate connection for Scheduler.retrieveIntervalChainsAndRun(),
   144  	// and another connection for LogHook.send()
   145  	connConfig.MaxConns = int32(pge.Resource.CronWorkers) + int32(pge.Resource.IntervalWorkers) + 3
   146  	connConfig.ConnConfig.RuntimeParams["application_name"] = "pg_timetable"
   147  	connConfig.ConnConfig.OnNotice = func(_ *pgconn.PgConn, n *pgconn.Notice) {
   148  		pge.l.WithField("severity", n.Severity).WithField("notice", n.Message).Info("Notice received")
   149  	}
   150  	connConfig.AfterConnect = func(ctx context.Context, pgconn *pgx.Conn) (err error) {
   151  		pge.l.WithField("pid", pgconn.PgConn().PID()).
   152  			WithField("client", pge.ClientName).
   153  			Debug("Trying to get lock for the session")
   154  		if err = pge.TryLockClientName(ctx, pgconn); err != nil {
   155  			return err
   156  		}
   157  
   158  		_, err = pgconn.Exec(ctx, "LISTEN "+quoteIdent(pge.ClientName))
   159  		if pge.logTypeOID == InvalidOid {
   160  			err = pgconn.QueryRow(ctx, "select coalesce(to_regtype('timetable.log_type')::oid, 0)").Scan(&pge.logTypeOID)
   161  		}
   162  		pgconn.TypeMap().RegisterType(&pgtype.Type{Name: "timetable.log_type", OID: pge.logTypeOID, Codec: &pgtype.EnumCodec{}})
   163  		return err
   164  	}
   165  	// reset session before returning connection to the pool
   166  	// after task completion, if the task is not properly finalized (especially when running in autonomous mode),
   167  	// some objects and/or setting changes will still exist in the session
   168  	connConfig.AfterRelease = func(pgconn *pgx.Conn) bool {
   169  		var err error
   170  		if pgconn.DeallocateAll(context.Background()) != nil {
   171  			return false // destroy the connection in case of error
   172  		}
   173  		if _, err = pgconn.Exec(context.Background(), "DISCARD ALL"); err == nil {
   174  			_, err = pgconn.Exec(context.Background(), "LISTEN "+quoteIdent(pge.ClientName))
   175  		}
   176  		return err == nil
   177  	}
   178  
   179  	if !pge.Start.Debug { //will handle notification in HandleNotifications directly
   180  		connConfig.ConnConfig.OnNotification = pge.NotificationHandler
   181  	}
   182  	tracelogger := &tracelog.TraceLog{
   183  		Logger:   log.NewPgxLogger(pge.l),
   184  		LogLevel: map[bool]tracelog.LogLevel{false: tracelog.LogLevelWarn, true: tracelog.LogLevelDebug}[pge.Verbose()],
   185  	}
   186  	connConfig.ConnConfig.Tracer = tracelogger
   187  	return connConfig
   188  }

   200  // TryLockClientName obtains lock on the server to prevent another client with the same name
   201  func (pge *PgEngine) TryLockClientName(ctx context.Context, conn QueryRowIface) error {
   202  	sql := "SELECT COALESCE(to_regproc('timetable.try_lock_client_name')::int4, 0)"
   203  	var procoid int // check if the schema is available first
   204  	if err := conn.QueryRow(ctx, sql).Scan(&procoid); err != nil {
   205  		return err
   206  	}
   207  	if procoid == 0 { //there is no schema yet, will lock after bootstrapping
   208  		pge.l.Debug("There is no schema yet, will lock after bootstrapping")
   209  		return nil
   210  	}
   211  	sql = "SELECT timetable.try_lock_client_name($1, $2)"
   212  	var locked bool
   213  	if e := conn.QueryRow(ctx, sql, pge.Getsid(), pge.ClientName).Scan(&locked); e != nil {
   214  		return e
   215  	} else if !locked {
   216  		return errors.New("cannot obtain lock for a session")
   217  	}
   218  	return nil
   219  }

   221  // ExecuteFileScript handles both SQL and YAML files based on file extension
   222  func (pge *PgEngine) ExecuteFileScript(ctx context.Context, cmdOpts config.CmdOptions) error {
   223  	filePath := cmdOpts.Start.File
   224  
   225  	// Determine file type by extension
   226  	fileExt := strings.ToLower(filepath.Ext(filePath))
   227  
   228  	switch fileExt {
   229  	case ".yaml", ".yml":
   230  		// Handle YAML chain definition files
   231  		if cmdOpts.Start.Validate {
   232  			// Only validate, don't import
   233  			_, err := ParseYamlFile(filePath)
   234  			if err != nil {
   235  				pge.l.WithError(err).Error("YAML validation failed")
   236  				return err
   237  			}
   238  			pge.l.WithField("file", filePath).Info("YAML file validation successful")
   239  			return nil
   240  		}
   241  
   242  		// Import YAML chains
   243  		return pge.LoadYamlChains(ctx, filePath, cmdOpts.Start.Replace)
   244  
   245  	case ".sql":
   246  		// Handle SQL script files (existing behavior)
   247  		return pge.ExecuteCustomScripts(ctx, filePath)
   248  
   249  	default:
   250  		// Try to detect content type for files without extension
   251  		content, err := os.ReadFile(filePath)
   252  		if err != nil {
   253  			pge.l.WithError(err).Error("cannot read file")
   254  			return err
   255  		}
   256  
   257  		// Check if it looks like YAML (starts with "chains:" or contains YAML markers)
   258  		contentStr := strings.TrimSpace(string(content))
   259  		if strings.HasPrefix(contentStr, "chains:") {
   260  			pge.l.WithField("file", filePath).Info("Detected YAML content, processing as YAML")
   261  			return pge.LoadYamlChains(ctx, filePath, false)
   262  		}
   263  		pge.l.WithField("file", filePath).Info("Processing as SQL script")
   264  		return pge.ExecuteCustomScripts(ctx, filePath)
   265  	}
   266  }

   268  // ExecuteCustomScripts executes SQL scripts in files
   269  func (pge *PgEngine) ExecuteCustomScripts(ctx context.Context, filename ...string) error {
   270  	for _, f := range filename {
   271  		sql, err := os.ReadFile(f)
   272  		if err != nil {
   273  			pge.l.WithError(err).Error("cannot read command file")
   274  			return err
   275  		}
   276  		pge.l.Info("Executing script: ", f)
   277  		if _, err = pge.ConfigDb.Exec(ctx, string(sql)); err != nil {
   278  			pge.l.WithError(err).Error("script execution failed")
   279  			return err
   280  		}
   281  		pge.l.Info("Script file executed: ", f)
   282  	}
   283  	return nil
   284  }

   286  // ExecuteSchemaScripts executes initial schema scripts
   287  func (pge *PgEngine) ExecuteSchemaScripts(ctx context.Context) error {
   288  	var exists bool
   289  	err := pge.ConfigDb.QueryRow(ctx, "SELECT EXISTS(SELECT 1 FROM pg_namespace WHERE nspname = 'timetable')").Scan(&exists)
   290  	if err != nil {
   291  		return err
   292  	}
   293  	if !exists {
   294  		for i, sql := range sqls {
   295  			sqlName := sqlNames[i]
   296  			pge.l.Info("Executing script: ", sqlName)
   297  			if _, err = pge.ConfigDb.Exec(ctx, sql); err != nil {
   298  				pge.l.WithError(err).Error("Script execution failed")
   299  				pge.l.Warn("Dropping \"timetable\" schema...")
   300  				_, err = pge.ConfigDb.Exec(ctx, "DROP SCHEMA IF EXISTS timetable CASCADE")
   301  				if err != nil {
   302  					pge.l.WithError(err).Error("Schema dropping failed")
   303  				}
   304  				return err
   305  			}
   306  			pge.l.Info("Schema file executed: " + sqlName)
   307  		}
   308  		pge.l.Info("Configuration schema created...")
   309  	}
   310  	return nil
   311  }
