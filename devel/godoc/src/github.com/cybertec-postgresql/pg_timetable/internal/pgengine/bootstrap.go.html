    80  func New(ctx context.Context, cmdOpts config.CmdOptions, logger log.LoggerHookerIface) (*PgEngine, error) {
    81  	var (
    82  		err error
    83  	)
    84  	pge := &PgEngine{
    85  		l:               logger,
    86  		ConfigDb:        nil,
    87  		CmdOptions:      cmdOpts,
    88  		chainSignalChan: make(chan ChainSignal, 64),
    89  	}
    90  	pge.l.WithField("sid", pge.Getsid()).Info("Starting new session... ")
    91  	config := pge.getPgxConnConfig()
    92  	if err = retry.Do(ctx, backoff, func(ctx context.Context) error {
    93  		if pge.ConfigDb, err = pgxpool.NewWithConfig(ctx, config); err == nil {
    94  			err = pge.ConfigDb.Ping(ctx)
    95  		}
    96  		if err != nil {
    97  			pge.l.WithError(err).Error("Connection failed")
    98  			pge.l.Info("Sleeping before reconnecting...")
    99  			return retry.RetryableError(err)
   100  		}
   101  		return nil
   102  	}); err != nil {
   103  		return nil, err
   104  	}
   105  	pge.l.Info("Database connection established")
   106  	if err := pge.ExecuteSchemaScripts(ctx); err != nil {
   107  		return nil, err
   108  	}
   109  	pge.AddLogHook(ctx) //schema exists, we can log now
   110  	if cmdOpts.Start.File != "" {
   111  		if err := pge.ExecuteCustomScripts(ctx, cmdOpts.Start.File); err != nil {
   112  			return nil, err
   113  		}
   114  	}
   115  	return pge, nil

   134  func (pge *PgEngine) getPgxConnConfig() *pgxpool.Config {
   135  	connConfig, err := pgxpool.ParseConfig(pge.ConnStr)
   136  	if err != nil {
   137  		pge.l.WithError(err).Error("Cannot parse connection string")
   138  		return nil
   139  	}
   140  	// in the worst scenario we need separate connections for each of workers,
   141  	// separate connection for Scheduler.retrieveChainsAndRun(),
   142  	// separate connection for Scheduler.retrieveIntervalChainsAndRun(),
   143  	// and another connection for LogHook.send()
   144  	connConfig.MaxConns = int32(pge.Resource.CronWorkers) + int32(pge.Resource.IntervalWorkers) + 3
   145  	connConfig.ConnConfig.RuntimeParams["application_name"] = "pg_timetable"
   146  	connConfig.ConnConfig.OnNotice = func(_ *pgconn.PgConn, n *pgconn.Notice) {
   147  		pge.l.WithField("severity", n.Severity).WithField("notice", n.Message).Info("Notice received")
   148  	}
   149  	connConfig.AfterConnect = func(ctx context.Context, pgconn *pgx.Conn) (err error) {
   150  		pge.l.WithField("pid", pgconn.PgConn().PID()).
   151  			WithField("client", pge.ClientName).
   152  			Debug("Trying to get lock for the session")
   153  		if err = pge.TryLockClientName(ctx, pgconn); err != nil {
   154  			return err
   155  		}
   156  
   157  		_, err = pgconn.Exec(ctx, "LISTEN "+quoteIdent(pge.ClientName))
   158  		if pge.logTypeOID == InvalidOid {
   159  			err = pgconn.QueryRow(ctx, "select coalesce(to_regtype('timetable.log_type')::oid, 0)").Scan(&pge.logTypeOID)
   160  		}
   161  		pgconn.TypeMap().RegisterType(&pgtype.Type{Name: "timetable.log_type", OID: pge.logTypeOID, Codec: &pgtype.EnumCodec{}})
   162  		return err
   163  	}
   164  	// reset session before returning connection to the pool
   165  	// after task completion, if the task is not properly finalized (especially when running in autonomous mode),
   166  	// some objects and/or setting changes will still exist in the session
   167  	connConfig.AfterRelease = func(pgconn *pgx.Conn) bool {
   168  		var err error
   169  		if pgconn.DeallocateAll(context.Background()) != nil {
   170  			return false // destroy the connection in case of error
   171  		}
   172  		if _, err = pgconn.Exec(context.Background(), "DISCARD ALL"); err == nil {
   173  			_, err = pgconn.Exec(context.Background(), "LISTEN "+quoteIdent(pge.ClientName))
   174  		}
   175  		return err == nil
   176  	}
   177  
   178  	if !pge.Start.Debug { //will handle notification in HandleNotifications directly
   179  		connConfig.ConnConfig.OnNotification = pge.NotificationHandler
   180  	}
   181  	tracelogger := &tracelog.TraceLog{
   182  		Logger:   log.NewPgxLogger(pge.l),
   183  		LogLevel: map[bool]tracelog.LogLevel{false: tracelog.LogLevelWarn, true: tracelog.LogLevelDebug}[pge.Verbose()],
   184  	}
   185  	connConfig.ConnConfig.Tracer = tracelogger
   186  	return connConfig
   187  }

   199  // TryLockClientName obtains lock on the server to prevent another client with the same name
   200  func (pge *PgEngine) TryLockClientName(ctx context.Context, conn QueryRowIface) error {
   201  	sql := "SELECT COALESCE(to_regproc('timetable.try_lock_client_name')::int4, 0)"
   202  	var procoid int // check if the schema is available first
   203  	if err := conn.QueryRow(ctx, sql).Scan(&procoid); err != nil {
   204  		return err
   205  	}
   206  	if procoid == 0 { //there is no schema yet, will lock after bootstrapping
   207  		pge.l.Debug("There is no schema yet, will lock after bootstrapping")
   208  		return nil
   209  	}
   210  	sql = "SELECT timetable.try_lock_client_name($1, $2)"
   211  	var locked bool
   212  	if e := conn.QueryRow(ctx, sql, pge.Getsid(), pge.ClientName).Scan(&locked); e != nil {
   213  		return e
   214  	} else if !locked {
   215  		return errors.New("cannot obtain lock for a session")
   216  	}
   217  	return nil
   218  }

   220  // ExecuteCustomScripts executes SQL scripts in files
   221  func (pge *PgEngine) ExecuteCustomScripts(ctx context.Context, filename ...string) error {
   222  	for _, f := range filename {
   223  		sql, err := os.ReadFile(f)
   224  		if err != nil {
   225  			pge.l.WithError(err).Error("cannot read command file")
   226  			return err
   227  		}
   228  		pge.l.Info("Executing script: ", f)
   229  		if _, err = pge.ConfigDb.Exec(ctx, string(sql)); err != nil {
   230  			pge.l.WithError(err).Error("script execution failed")
   231  			return err
   232  		}
   233  		pge.l.Info("Script file executed: ", f)
   234  	}
   235  	return nil
   236  }

   238  // ExecuteSchemaScripts executes initial schema scripts
   239  func (pge *PgEngine) ExecuteSchemaScripts(ctx context.Context) error {
   240  	var exists bool
   241  	err := pge.ConfigDb.QueryRow(ctx, "SELECT EXISTS(SELECT 1 FROM pg_namespace WHERE nspname = 'timetable')").Scan(&exists)
   242  	if err != nil {
   243  		return err
   244  	}
   245  	if !exists {
   246  		for i, sql := range sqls {
   247  			sqlName := sqlNames[i]
   248  			pge.l.Info("Executing script: ", sqlName)
   249  			if _, err = pge.ConfigDb.Exec(ctx, sql); err != nil {
   250  				pge.l.WithError(err).Error("Script execution failed")
   251  				pge.l.Warn("Dropping \"timetable\" schema...")
   252  				_, err = pge.ConfigDb.Exec(ctx, "DROP SCHEMA IF EXISTS timetable CASCADE")
   253  				if err != nil {
   254  					pge.l.WithError(err).Error("Schema dropping failed")
   255  				}
   256  				return err
   257  			}
   258  			pge.l.Info("Schema file executed: " + sqlName)
   259  		}
   260  		pge.l.Info("Configuration schema created...")
   261  	}
   262  	return nil
   263  }
