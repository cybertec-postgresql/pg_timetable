    61  	mail.SetHeader("Cc", ccrecipients...)
    62  
    63  	// Multiple bCC Addresses
    64  	bccrecipients := make([]string, len(conn.BccAddr))
    65  	for i, bccaddr := range conn.BccAddr {
    66  		bccrecipients[i] = mail.FormatAddress(bccaddr, " ")
    67  	}
    68  	mail.SetHeader("Bcc", bccrecipients...)
    69  
    70  	mail.SetHeader("Subject", conn.Subject)
    71  	mail.SetBody(conn.ContentType, conn.MsgBody)
    72  
    73  	//attach multiple documents
    74  	for _, attachment := range conn.Attachments {
    75  		mail.Attach(attachment)
    76  	}
    77  
    78  	// Attach file with contents
    79  	for _, attachmentData := range conn.AttachmentData {

    81  	}
    82  	// Send Mail
