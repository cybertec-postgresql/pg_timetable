    29  	// For slice-valued flags (e.g. --file), fmt's default formatting would
    30  	// produce a bracket-wrapped string like "[a b]", which viper then stores
    31  	// as a single string. Join with the separator viper uses to split string

    93  func NewConfig(writer io.Writer) (*CmdOptions, error) {
    94  	v := viper.New()
    95  	p, err := Parse(writer)
    96  	if err != nil {
    97  		return nil, err
    98  	}
    99  	flagSet := cmdArgSet{p}
   100  	if err = v.BindFlagValues(flagSet); err != nil {
   101  		return nil, fmt.Errorf("cannot bind command-line flag values with viper: %w", err)
   102  	}
   103  	flagSet.setDefaults(v)
   104  	if v.IsSet("config") {
   105  		v.SetConfigFile(v.GetString("config"))
   106  		err := v.ReadInConfig() // Find and read the config file
   107  		if err != nil {         // Handle errors reading the config file
   108  			return nil, fmt.Errorf("fatal error reading config file: %w", err)
   109  		}
   110  	}
   111  	conf := &CmdOptions{}
   112  	if err = v.Unmarshal(conf); err != nil {
   113  		return nil, fmt.Errorf("fatal error unmarshalling config file: %w", err)
   114  	}
   115  	// viper may bind the default value of the []string `file` flag as a single
   116  	// empty string ([""]) when no --file is provided. Strip empty entries so
   117  	// startup file processing is not triggered for non-existent paths.
   118  	conf.Start.File = filterEmpty(conf.Start.File)
   119  	if conf.ClientName == "" {
   120  		buf := bytes.NewBufferString("The required flag `-c, --clientname` was not specified\n")
   121  		p.WriteHelp(buf)
   122  		return conf, errors.New(buf.String())
   123  	}
   124  	if err := ValidateOTel(conf.OTel); err != nil {
   125  		return conf, err
   126  	}
   127  	return conf, nil
   128  }

   130  // filterEmpty returns a new slice with blank (empty or whitespace-only)

   132  func filterEmpty(in []string) []string {
   133  	out := in[:0]
   134  	for _, s := range in {
   135  		if s = strings.TrimSpace(s); s != "" {
   136  			out = append(out, s)

   138  	}

   143  func ValidateOTel(opts OTelOpts) error {

   158  		switch u.Scheme {

   163  		}
   164  	}
