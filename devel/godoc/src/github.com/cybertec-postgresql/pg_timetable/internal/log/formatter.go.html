    52  func (f *Formatter) Format(entry *logrus.Entry) ([]byte, error) {
    53  	levelColor := getColorByLevel(entry.Level)
    54  
    55  	timestampFormat := f.TimestampFormat
    56  	if timestampFormat == "" {
    57  		timestampFormat = time.StampMilli
    58  	}
    59  
    60  	// output buffer
    61  	b := &bytes.Buffer{}
    62  
    63  	// write time
    64  	b.WriteString(entry.Time.Format(timestampFormat))
    65  
    66  	// write level
    67  	var level string
    68  	if f.NoUppercaseLevel {
    69  		level = entry.Level.String()
    70  	} else {
    71  		level = strings.ToUpper(entry.Level.String())
    72  	}
    73  
    74  	if f.CallerFirst {
    75  		f.writeCaller(b, entry)
    76  	}
    77  
    78  	if !f.NoColors {
    79  		fmt.Fprintf(b, "\x1b[%dm", levelColor)
    80  	}
    81  
    82  	b.WriteString(" [")
    83  	if f.ShowFullLevel {
    84  		b.WriteString(level)
    85  	} else {
    86  		b.WriteString(level[:4])
    87  	}
    88  	b.WriteString("]")
    89  
    90  	if !f.NoFieldsSpace {
    91  		b.WriteString(" ")
    92  	}
    93  
    94  	if !f.NoColors && f.NoFieldsColors {
    95  		b.WriteString("\x1b[0m")
    96  	}
    97  
    98  	// write fields
    99  	if f.FieldsOrder == nil {
   100  		f.writeFields(b, entry)
   101  	} else {
   102  		f.writeOrderedFields(b, entry)
   103  	}
   104  
   105  	if f.NoFieldsSpace {
   106  		b.WriteString(" ")
   107  	}
   108  
   109  	if !f.NoColors && !f.NoFieldsColors {
   110  		b.WriteString("\x1b[0m")
   111  	}
   112  
   113  	// write message
   114  	if f.TrimMessages {
   115  		b.WriteString(strings.TrimSpace(entry.Message))
   116  	} else {
   117  		b.WriteString(entry.Message)
   118  	}
   119  
   120  	if !f.CallerFirst {
   121  		f.writeCaller(b, entry)
   122  	}
   123  
   124  	b.WriteByte('\n')
   125  
   126  	return b.Bytes(), nil

   138  func (f *Formatter) writeCaller(b *bytes.Buffer, entry *logrus.Entry) {

   157  func (f *Formatter) writeFields(b *bytes.Buffer, entry *logrus.Entry) {
   158  	if len(entry.Data) != 0 {
   159  		fields := make([]string, 0, len(entry.Data))
   160  		for field := range entry.Data {
   161  			fields = append(fields, field)
   162  		}
   163  
   164  		sort.Strings(fields)
   165  
   166  		for _, field := range fields {

   172  func (f *Formatter) writeOrderedFields(b *bytes.Buffer, entry *logrus.Entry) {
   173  	length := len(entry.Data)
   174  	foundFieldsMap := map[string]bool{}
   175  	for _, field := range f.FieldsOrder {
   176  		if _, ok := entry.Data[field]; ok {
   177  			foundFieldsMap[field] = true
   178  			length--
   179  			f.writeField(b, entry, field)
   180  		}
   181  	}
   182  
   183  	if length > 0 {
   184  		notFoundFields := make([]string, 0, length)
   185  		for field := range entry.Data {
   186  			if !foundFieldsMap[field] {
   187  				notFoundFields = append(notFoundFields, field)
   188  			}
   189  		}
   190  
   191  		sort.Strings(notFoundFields)
   192  
   193  		for _, field := range notFoundFields {
   194  			f.writeField(b, entry, field)
   195  		}
   196  	}
   197  }

   199  func (f *Formatter) writeField(b *bytes.Buffer, entry *logrus.Entry, field string) {
   200  	if f.HideKeys {

   203  		fmt.Fprintf(b, "[%s:%v]", field, entry.Data[field])
   204  	}
   205  
   206  	if !f.NoFieldsSpace {
   207  		b.WriteString(" ")
   208  	}
