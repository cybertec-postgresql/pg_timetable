    77  // Log transforms logging calls from pgx to logrus
    78  func (pgxlogger *PgxLogger) Log(ctx context.Context, level tracelog.LogLevel, msg string, data map[string]any) {
    79  	logger := GetLogger(ctx)
    80  	if logger == FallbackLogger { //switch from standard to specified
    81  		logger = pgxlogger.l
    82  	}
    83  	if data != nil {
    84  		logger = logger.WithFields(data)
    85  	}
    86  	switch level {
    87  	case tracelog.LogLevelTrace:
    88  		logger.WithField("PGX_LOG_LEVEL", level).Debug(msg)
    89  	case tracelog.LogLevelDebug, tracelog.LogLevelInfo: //pgx is way too chatty on INFO level
    90  		logger.Debug(msg)
    91  	case tracelog.LogLevelWarn:
    92  		logger.Warn(msg)
    93  	case tracelog.LogLevelError:
    94  		logger.Error(msg)
    95  	default:
    96  		logger.WithField("INVALID_PGX_LOG_LEVEL", level).Error(msg)
    97  	}
    98  }
