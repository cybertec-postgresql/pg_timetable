    45  // New initialises an OTel Provider from the given options.
    46  // When opts.Endpoint is empty or both Traces and Metrics are false, a noop
    47  // Provider is returned without connecting to any backend.
    48  func New(ctx context.Context, opts config.OTelOpts, clientName, serviceVersion string) (*Provider, error) {
    49  	p := &Provider{
    50  		shutdownTimeout: time.Duration(opts.ShutdownTimeout) * time.Second,
    51  	}
    52  	if opts.Endpoint == "" || (!opts.Traces && !opts.Metrics) {
    53  		return p, nil
    54  	}
    55  
    56  	res, err := buildResource(ctx, opts.ServiceName, clientName, serviceVersion)
    57  	if err != nil {
    58  		return p, fmt.Errorf("otel: build resource: %w", err)
    59  	}
    60  
    61  	u, err := url.Parse(opts.Endpoint)
    62  	if err != nil {
    63  		return nil, fmt.Errorf("otel: parse endpoint: %w", err)
    64  	}
    65  
    66  	if opts.Traces {
    67  		tp, err := buildTracerProvider(ctx, u, opts, res)
    68  		if err != nil {
    69  			return nil, err
    70  		}
    71  		p.tp = tp
    72  	}
    73  
    74  	if opts.Metrics {
    75  		mp, err := buildMeterProvider(ctx, u, opts, res)
    76  		if err != nil {
    77  			return nil, err
    78  		}
    79  		p.mp = mp
    80  		if err := p.registerMetrics(); err != nil {
    81  			return nil, err
    82  		}
    83  	}
    84  
    85  	return p, nil
    86  }

   112  // Shutdown flushes pending spans and metrics and releases resources.
   113  // It is safe to call multiple times.
   114  func (p *Provider) Shutdown(ctx context.Context) error {
   115  	var errs []error
   116  	if p.tp != nil {
   117  		if err := p.tp.Shutdown(ctx); err != nil {
   118  			errs = append(errs, err)
   119  		}
   120  	}
   121  	if p.mp != nil {
   122  		if err := p.mp.Shutdown(ctx); err != nil {
   123  			errs = append(errs, err)
   124  		}
   125  	}
   126  	if len(errs) > 0 {
   127  		return fmt.Errorf("otel shutdown: %v", errs)
   128  	}
   129  	return nil
   130  }

   147  func buildTracerProvider(ctx context.Context, u *url.URL, opts config.OTelOpts, res *resource.Resource) (*sdktrace.TracerProvider, error) {
   148  	exp, err := buildTraceExporter(ctx, u, opts)
   149  	if err != nil {
   150  		return nil, err

   152  	sampler := sdktrace.ParentBased(sdktrace.TraceIDRatioBased(opts.SampleRatio))
   153  	tp := sdktrace.NewTracerProvider(
   154  		sdktrace.WithBatcher(exp),
   155  		sdktrace.WithSampler(sampler),
   156  		sdktrace.WithResource(res),
   157  	)
   158  	return tp, nil

   161  func buildTraceExporter(ctx context.Context, u *url.URL, opts config.OTelOpts) (sdktrace.SpanExporter, error) {
   162  	switch u.Scheme {
   163  	case "grpc":
   164  		grpcOpts := []otlptracegrpc.Option{
   165  			otlptracegrpc.WithEndpoint(u.Host),
   166  			otlptracegrpc.WithHeaders(opts.Headers),
   167  		}
   168  		if opts.Insecure {
   169  			grpcOpts = append(grpcOpts, otlptracegrpc.WithInsecure())
   170  		}
   171  		return otlptracegrpc.New(ctx, grpcOpts...)
   172  	case "http", "https":
   173  		httpOpts := []otlptracehttp.Option{
   174  			otlptracehttp.WithEndpoint(u.Host),
   175  			otlptracehttp.WithHeaders(opts.Headers),
   176  		}
   177  		if opts.Insecure {
   178  			httpOpts = append(httpOpts, otlptracehttp.WithInsecure())
   179  		}
   180  		return otlptracehttp.New(ctx, httpOpts...)
