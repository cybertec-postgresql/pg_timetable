   105  	// gRPC connections are lazy — New() should succeed even if backend is down

   107  	opts := config.OTelOpts{
   108  		Endpoint:        "grpc://localhost:19999",
   109  		Traces:          true,

   111  		SampleRatio:     1.0,
   112  		MetricPeriod:    30,
   113  		ShutdownTimeout: 1,
   114  		ServiceName:     "pg_timetable",
   115  	}
   116  	p, err := otel.New(ctx, opts, "test-client", "0.0.1")
   117  	// gRPC is lazy — no error expected at provider creation time
   118  	if err != nil {
   119  		t.Logf("Note: gRPC New() returned error (may be eager): %v", err)
   120  		// This is acceptable — main.go falls back to noop on error
   121  		return
   122  	}
   123  	assert.NotNil(t, p)
   124  	// Shutdown should complete within 1 second even with no backend
   125  	shutdownCtx, cancel := context.WithTimeout(context.Background(), 2*time.Second)
   126  	defer cancel()
   127  	_ = p.Shutdown(shutdownCtx) // errors acceptable here
   128  }
