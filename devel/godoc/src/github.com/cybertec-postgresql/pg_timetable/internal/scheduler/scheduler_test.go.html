    17  func TestExecuteChainCancelledContextCleansUp(t *testing.T) {
    18  	container, cleanup := testutils.SetupPostgresContainer(t)
    19  	defer cleanup()
    20  
    21  	pge := container.Engine
    22  	ctx := context.Background()
    23  	sch := New(pge, log.Init(config.LoggingOpts{LogLevel: "panic", LogDBLevel: "none"}), otel.NewNoop())
    24  
    25  	var chainID int
    26  	err := pge.ConfigDb.QueryRow(ctx, `
    27  		INSERT INTO timetable.chain(chain_name, max_instances)
    28  		VALUES ('cancel_cleanup_test', 16) RETURNING chain_id`).Scan(&chainID)
    29  	require.NoError(t, err)
    30  
    31  	_, err = pge.ConfigDb.Exec(ctx, `
    32  		INSERT INTO timetable.task(chain_id, task_order, kind, command)
    33  		VALUES ($1, 1, 'SQL', 'SELECT pg_sleep(0.5)')`, chainID)
    34  	require.NoError(t, err)
    35  
    36  	// Simulate what chainWorker does before calling executeChain.
    37  	require.True(t, pge.InsertChainRunStatus(ctx, chainID, 16), "chain run status should be inserted")
    38  
    39  	// Cancel the context while the task is sleeping, mimicking notify_chain_stop().
    40  	execCtx, cancel := context.WithCancel(ctx)
    41  	go func() {
    42  		time.Sleep(100 * time.Millisecond)
    43  		cancel()
    44  	}()
    45  
    46  	sch.executeChain(execCtx, Chain{ChainID: chainID, MaxInstances: 16})
    47  
    48  	// The stale active_chain row must be gone: cleanup must succeed despite the
    49  	// cancelled context.
    50  	var count int
    51  	err = pge.ConfigDb.QueryRow(ctx,
    52  		"SELECT count(*) FROM timetable.active_chain WHERE chain_id = $1", chainID).Scan(&count)
    53  	require.NoError(t, err)
    54  	assert.Zero(t, count, "active_chain entry must be removed even when the execution context was cancelled")
    55  }
