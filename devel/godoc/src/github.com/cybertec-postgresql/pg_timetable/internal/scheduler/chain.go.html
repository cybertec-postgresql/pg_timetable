    61  func (sch *Scheduler) processAsyncChain(ctx context.Context, chainSignal ChainSignal) error {
    62  	switch chainSignal.Command {
    63  	case "START":
    64  		var c Chain
    65  		if err := sch.pgengine.SelectChain(ctx, &c, chainSignal.ConfigID); err != nil {
    66  			return fmt.Errorf("cannot start chain with ID: %d; %w", chainSignal.ConfigID, err)
    67  		}
    68  		go func() {
    69  			select {
    70  			case <-ctx.Done():
    71  				return
    72  			case <-time.After(time.Duration(chainSignal.Delay) * time.Second):
    73  				sch.SendChain(c)
    74  			}
    75  		}()
    76  	case "STOP":
    77  		if cancel, ok := sch.activeChains[chainSignal.ConfigID]; ok {
    78  			cancel()
    79  			return nil
    80  		}
    81  		return fmt.Errorf("cannot stop chain with ID: %d. No running chain found", chainSignal.ConfigID)
    82  	}
    83  	return nil
    84  }

   102  	headChainsCount := len(headChains)
   103  	sch.l.WithField("count", headChainsCount).Info(msg)
   104  	// now we can loop through the chains
   105  	for _, c := range headChains {
   106  		// if the number of chains pulled for execution is high, try to spread execution to avoid spikes
   107  		if headChainsCount > sch.Config().Resource.CronWorkers*refetchTimeout {
   108  			time.Sleep(time.Duration(refetchTimeout*1000/headChainsCount) * time.Millisecond)
   109  		}
   110  		sch.SendChain(c)
   111  	}

   140  func (sch *Scheduler) chainWorker(ctx context.Context, chains <-chan Chain) {
   141  	for {
   142  		select {
   143  		case <-ctx.Done(): //check context with high priority
   144  			return
   145  		default:
   146  			select {
   147  			case chain := <-chains:
   148  				chainL := sch.l.WithField("chain", chain.ChainID)
   149  				chainContext := log.WithLogger(ctx, chainL)
   150  				if !sch.pgengine.InsertChainRunStatus(ctx, chain.ChainID, chain.MaxInstances) {
   151  					chainL.Info("Cannot proceed. Sleeping")
   152  					continue
   153  				}
   154  				chainL.Info("Starting chain")
   155  				sch.Lock(chain.ExclusiveExecution)
   156  				chainContext, cancel := context.WithCancel(chainContext)
   157  				sch.addActiveChain(chain.ChainID, cancel)
   158  				sch.executeChain(chainContext, chain)
   159  				sch.deleteActiveChain(chain.ChainID)
   160  				cancel()
   161  				sch.Unlock(chain.ExclusiveExecution)
   162  			case <-ctx.Done():
   163  				return
   164  			}
   165  
   166  		}
   167  	}

   192  func (sch *Scheduler) executeChain(ctx context.Context, chain Chain) {
   193  	var ChainTasks []pgengine.ChainTask
   194  	var bctx context.Context
   195  	var cancel context.CancelFunc
   196  	var vxid int64
   197  
   198  	chainCtx, cancel := getTimeoutContext(ctx, sch.Config().Resource.ChainTimeout, chain.Timeout)
   199  	if cancel != nil {
   200  		defer cancel()
   201  	}
   202  
   203  	chainL := sch.l.WithField("chain", chain.ChainID)
   204  	tx, vxid, err := sch.pgengine.StartTransaction(chainCtx)
   205  	if err != nil {
   206  		chainL.WithError(err).Error("Cannot start transaction")
   207  		return
   208  	}
   209  	chainL = chainL.WithField("vxid", vxid)
   210  
   211  	err = sch.pgengine.GetChainElements(chainCtx, &ChainTasks, chain.ChainID)
   212  	if err != nil {
   213  		chainL.WithError(err).Error("Failed to retrieve chain elements")
   214  		sch.pgengine.RollbackTransaction(chainCtx, tx)
   215  		return
   216  	}
   217  
   218  	/* now we can loop through every element of the task chain */
   219  	for _, task := range ChainTasks {
   220  		task.ChainID = chain.ChainID
   221  		task.Vxid = vxid
   222  		l := chainL.WithField("task", task.TaskID)
   223  		l.Info("Starting task")
   224  		taskCtx := log.WithLogger(chainCtx, l)
   225  		retCode := sch.executeTask(taskCtx, tx, &task)
   226  
   227  		// we use background context here because current one (chainCtx) might be cancelled
   228  		bctx = log.WithLogger(ctx, l)
   229  		if retCode != 0 {
   230  			if !task.IgnoreError {
   231  				chainL.Error("Chain failed")
   232  				sch.pgengine.RemoveChainRunStatus(bctx, chain.ChainID)
   233  				sch.pgengine.RollbackTransaction(bctx, tx)
   234  				sch.executeOnErrorHandler(bctx, chain)
   235  				return
   236  			}
   237  			l.Info("Ignoring task failure")
   238  		}
   239  	}
   240  	bctx = log.WithLogger(chainCtx, chainL)
   241  	sch.pgengine.CommitTransaction(bctx, tx)
   242  	chainL.Info("Chain executed successfully")
   243  	sch.pgengine.RemoveChainRunStatus(bctx, chain.ChainID)
   244  	if chain.SelfDestruct {
   245  		sch.pgengine.DeleteChain(bctx, chain.ChainID)
   246  	}
   247  }

   250  func (sch *Scheduler) executeTask(ctx context.Context, tx pgx.Tx, task *pgengine.ChainTask) int {
   251  	var (
   252  		paramValues []string
   253  		err         error
   254  		out         string
   255  		retCode     int
   256  		cancel      context.CancelFunc
   257  	)
   258  
   259  	l := log.GetLogger(ctx)
   260  	err = sch.pgengine.GetChainParamValues(ctx, &paramValues, task)
   261  	if err != nil {
   262  		l.WithError(err).Error("cannot fetch parameters values for chain: ", err)
   263  		return -1
   264  	}
   265  
   266  	ctx, cancel = getTimeoutContext(ctx, sch.Config().Resource.TaskTimeout, task.Timeout)
   267  	if cancel != nil {
   268  		defer cancel()
   269  	}
   270  
   271  	task.StartedAt = time.Now()
   272  	switch task.Kind {
   273  	case "SQL":
   274  		out, err = sch.pgengine.ExecuteSQLTask(ctx, tx, task, paramValues)
   275  	case "PROGRAM":
   276  		if sch.pgengine.NoProgramTasks {
   277  			l.Info("Program task execution skipped")
   278  			return -2
   279  		}
   280  		retCode, out, err = sch.ExecuteProgramCommand(ctx, task.Script, paramValues)
   281  	case "BUILTIN":
   282  		out, err = sch.executeBuiltinTask(ctx, task.Script, paramValues)
   283  	}
   284  	task.Duration = time.Since(task.StartedAt).Microseconds()
   285  
   286  	if err != nil {
   287  		if retCode == 0 {
   288  			retCode = -1
   289  		}
   290  		out = strings.Join([]string{out, err.Error()}, "\n")
   291  		l.WithError(err).Error("Task execution failed")
   292  	} else {
   293  		l.Info("Task executed successfully")
   294  	}
   295  	sch.pgengine.LogTaskExecution(context.Background(), task, retCode, out)
   296  	return retCode
   297  }
