    64  func (sch *Scheduler) processAsyncChain(ctx context.Context, chainSignal ChainSignal) error {
    65  	switch chainSignal.Command {
    66  	case "START":
    67  		var c Chain
    68  		if err := sch.pgengine.SelectChain(ctx, &c, chainSignal.ConfigID); err != nil {
    69  			return fmt.Errorf("cannot start chain with ID: %d; %w", chainSignal.ConfigID, err)
    70  		}
    71  		go func() {
    72  			select {
    73  			case <-ctx.Done():
    74  				return
    75  			case <-time.After(time.Duration(chainSignal.Delay) * time.Second):
    76  				sch.SendChain(c)
    77  			}
    78  		}()
    79  	case "STOP":
    80  		if cancel, ok := sch.activeChains[chainSignal.ConfigID]; ok {
    81  			cancel()
    82  			return nil
    83  		}
    84  		return fmt.Errorf("cannot stop chain with ID: %d. No running chain found", chainSignal.ConfigID)
    85  	}
    86  	return nil
    87  }

    89  func (sch *Scheduler) retrieveChainsAndRun(ctx context.Context, reboot bool) {
    90  	var err error
    91  	var headChains []Chain
    92  	msg := "Retrieve scheduled chains to run"
    93  	if reboot {
    94  		msg += " @reboot"
    95  	}
    96  	if reboot {
    97  		err = sch.pgengine.SelectRebootChains(ctx, &headChains)
    98  	} else {
    99  		err = sch.pgengine.SelectChains(ctx, &headChains)
   100  	}
   101  	if err != nil {
   102  		sch.l.WithError(err).Error("Could not query pending tasks")
   103  		return
   104  	}
   105  	headChainsCount := len(headChains)
   106  	sch.l.WithField("count", headChainsCount).Info(msg)
   107  	// now we can loop through the chains
   108  	for _, c := range headChains {
   109  		// if the number of chains pulled for execution is high, try to spread execution to avoid spikes
   110  		if headChainsCount > sch.Config().Resource.CronWorkers*refetchTimeout {
   111  			time.Sleep(time.Duration(refetchTimeout*1000/headChainsCount) * time.Millisecond)
   112  		}
   113  		sch.SendChain(c)
   114  	}
   115  }

   143  func (sch *Scheduler) chainWorker(ctx context.Context, chains <-chan Chain) {
   144  	for {
   145  		select {
   146  		case <-ctx.Done(): //check context with high priority
   147  			return
   148  		default:
   149  			select {
   150  			case chain := <-chains:
   151  				chainL := sch.l.WithField("chain", chain.ChainID)
   152  				chainContext := log.WithLogger(ctx, chainL)
   153  				if !sch.pgengine.InsertChainRunStatus(ctx, chain.ChainID, chain.MaxInstances) {
   154  					chainL.Info("Cannot proceed. Sleeping")
   155  					continue
   156  				}
   157  				chainL.Info("Starting chain")
   158  				sch.Lock(chain.ExclusiveExecution)
   159  				chainContext, cancel := context.WithCancel(chainContext)
   160  				sch.addActiveChain(chain.ChainID, cancel)
   161  				sch.executeChain(chainContext, chain)
   162  				sch.deleteActiveChain(chain.ChainID)
   163  				cancel()
   164  				sch.Unlock(chain.ExclusiveExecution)
   165  			case <-ctx.Done():
   166  				return
   167  			}
   168  
   169  		}
   170  	}
   171  }

   195  func (sch *Scheduler) executeChain(ctx context.Context, chain Chain) {
   196  	var ChainTasks []pgengine.ChainTask
   197  	var bctx context.Context
   198  	var cancel context.CancelFunc
   199  	var vxid int64
   200  
   201  	// OTel tracing: create root span for this chain execution
   202  	chainStart := time.Now()
   203  	tracer := sch.provider.Tracer()
   204  	ctx, chainSpan := tracer.Start(ctx, "chain.execute",
   205  		trace.WithAttributes(
   206  			attribute.Int("chain.id", chain.ChainID),
   207  			attribute.String("chain.name", chain.ChainName),
   208  			attribute.String("client.name", sch.Config().ClientName),
   209  		))
   210  	defer chainSpan.End()
   211  	sch.provider.RecordChainStarted(ctx, sch.Config().ClientName)
   212  
   213  	chainCtx, cancel := getTimeoutContext(ctx, sch.Config().Resource.ChainTimeout, chain.Timeout)
   214  	if cancel != nil {
   215  		defer cancel()
   216  	}
   217  
   218  	chainL := sch.l.WithField("chain", chain.ChainID)
   219  	tx, vxid, err := sch.pgengine.StartTransaction(chainCtx)
   220  	if err != nil {
   221  		chainL.WithError(err).Error("Cannot start transaction")
   222  		return
   223  	}
   224  	chainL = chainL.WithField("vxid", vxid)
   225  
   226  	err = sch.pgengine.GetChainElements(chainCtx, &ChainTasks, chain.ChainID)
   227  	if err != nil {
   228  		chainL.WithError(err).Error("Failed to retrieve chain elements")
   229  		sch.pgengine.RollbackTransaction(chainCtx, tx)
   230  		return
   231  	}
   232  
   233  	/* now we can loop through every element of the task chain */
   234  	for _, task := range ChainTasks {
   235  		task.ChainID = chain.ChainID
   236  		task.Vxid = vxid
   237  		l := chainL.WithField("task", task.TaskID)
   238  		l.Info("Starting task")
   239  		taskCtx := log.WithLogger(chainCtx, l)
   240  		err = sch.executeTask(taskCtx, tx, &task)
   241  		if err != nil {
   242  			l.WithError(err).Error("Task execution failed")
   243  		} else {
   244  			l.Info("Task executed successfully")
   245  		}
   246  
   247  		// we use background context here because current one (chainCtx) might be cancelled
   248  		bctx = log.WithLogger(ctx, l)
   249  		if err != nil {
   250  			if !task.IgnoreError {
   251  				chainL.Error("Chain failed")
   252  				sch.pgengine.RemoveChainRunStatus(bctx, chain.ChainID)
   253  				sch.pgengine.RollbackTransaction(bctx, tx)
   254  				chainSpan.SetStatus(codes.Error, "chain failed")
   255  				sch.provider.RecordChainFailed(bctx, sch.Config().ClientName)
   256  				sch.executeOnErrorHandler(bctx, chain)
   257  				return
   258  			}
   259  			l.Info("Ignoring task failure")
   260  		}
   261  	}
   262  	bctx = log.WithLogger(chainCtx, chainL)
   263  	sch.pgengine.CommitTransaction(bctx, tx)
   264  	sch.provider.RecordChainDuration(ctx, time.Since(chainStart).Seconds(), sch.Config().ClientName)
   265  	chainL.Info("Chain executed successfully")
   266  	sch.provider.RecordChainCompleted(ctx, sch.Config().ClientName)
   267  	sch.pgengine.RemoveChainRunStatus(bctx, chain.ChainID)
   268  	if chain.SelfDestruct {
   269  		sch.pgengine.DeleteChain(bctx, chain.ChainID)
   270  	}
   271  }

   273  /* execute a task */
   274  func (sch *Scheduler) executeTask(ctx context.Context, tx pgx.Tx, task *pgengine.ChainTask) error {
   275  	var (
   276  		paramValues []string
   277  		err         error
   278  		cancel      context.CancelFunc
   279  	)
   280  
   281  	// OTel tracing: create child span for this task execution
   282  	tracer := sch.provider.Tracer()
   283  	ctx, taskSpan := tracer.Start(ctx, "task.execute",
   284  		trace.WithAttributes(
   285  			attribute.String("task.name", task.Command),
   286  			attribute.String("task.kind", task.Kind),
   287  		))
   288  	defer taskSpan.End()
   289  
   290  	l := log.GetLogger(ctx)
   291  	err = sch.pgengine.GetChainParamValues(ctx, &paramValues, task)
   292  	if err != nil {
   293  		l.WithError(err).Error("cannot fetch parameters values for chain: ", err)
   294  		return err
   295  	}
   296  
   297  	ctx, cancel = getTimeoutContext(ctx, sch.Config().Resource.TaskTimeout, task.Timeout)
   298  	if cancel != nil {
   299  		defer cancel()
   300  	}
   301  
   302  	task.StartedAt = time.Now()
   303  	switch task.Kind {
   304  	case "SQL":
   305  		err = sch.pgengine.ExecuteSQLTask(ctx, tx, task, paramValues)
   306  	case "PROGRAM":
   307  		if sch.pgengine.NoProgramTasks {
   308  			l.Info("Program task execution skipped")
   309  			return errors.New("program tasks execution is disabled")
   310  		}
   311  		err = sch.ExecuteProgramCommand(ctx, task, paramValues)
   312  	case "BUILTIN":
   313  		err = sch.executeBuiltinTask(ctx, task, paramValues)
   314  	}
   315  	task.Duration = time.Since(task.StartedAt).Microseconds()
   316  	returnCode := 0
   317  	if err != nil {
   318  		returnCode = -1
   319  		taskSpan.RecordError(err)
   320  		taskSpan.SetStatus(codes.Error, err.Error())
   321  	}
   322  	taskSpan.SetAttributes(attribute.Int("task.return_code", returnCode))
   323  	sch.provider.RecordTaskExecuted(ctx, sch.Config().ClientName, task.Kind)
   324  	return err
   325  }
