    93  // Run executes jobs. Returns RunStatus why it terminated.
    94  // There are only two possibilities: dropped connection and cancelled context.
    95  func (sch *Scheduler) Run(ctx context.Context) RunStatus {
    96  	// create sleeping workers waiting data on channel
    97  	for w := 1; w <= sch.Config().Resource.CronWorkers; w++ {
    98  		workerCtx, cancel := context.WithCancel(ctx)
    99  		defer cancel()
   100  		go sch.chainWorker(workerCtx, sch.chainsChan)
   101  	}
   102  	for w := 1; w <= sch.Config().Resource.IntervalWorkers; w++ {
   103  		workerCtx, cancel := context.WithCancel(ctx)
   104  		defer cancel()
   105  		go sch.intervalChainWorker(workerCtx, sch.ichainsChan)
   106  	}
   107  	ctx = log.WithLogger(ctx, sch.l)
   108  
   109  	/*
   110  		Loop forever or until we ask it to stop.
   111  		First loop fetches notifications.
   112  		Main loop works every refetchTimeout seconds and runs chains.
   113  	*/
   114  	sch.l.Info("Accepting asynchronous chains execution requests...")
   115  	go sch.retrieveAsyncChainsAndRun(ctx)
   116  
   117  	if sch.Config().Start.Debug { //run blocking notifications receiving
   118  		sch.pgengine.HandleNotifications(ctx)
   119  		return ContextCancelledStatus
   120  	}
   121  
   122  	sch.l.Debug("Checking for @reboot task chains...")
   123  	sch.retrieveChainsAndRun(ctx, true)
   124  
   125  	// Use ticker for strict intervals
   126  	ticker := time.NewTicker(refetchTimeout * time.Second)
   127  	defer ticker.Stop()
   128  	for {
   129  		sch.l.Debug("Checking for task chains...")
   130  		go sch.retrieveChainsAndRun(ctx, false)
   131  		sch.l.Debug("Checking for interval task chains...")
   132  		go sch.retrieveIntervalChainsAndRun(ctx)
   133  
   134  		select {
   135  		case <-ticker.C:
   136  			// pass
   137  		case <-ctx.Done():
   138  			sch.status = ContextCancelledStatus
   139  		case <-sch.shutdown:
   140  			sch.status = ShutdownStatus
   141  			sch.terminateChains()
   142  		}
   143  		if sch.status != RunningStatus {
   144  			return sch.status
   145  		}
   146  	}
   147  }
