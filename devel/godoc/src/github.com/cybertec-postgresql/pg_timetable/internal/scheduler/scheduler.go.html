    96  // Run executes jobs. Returns RunStatus why it terminated.
    97  // There are only two possibilities: dropped connection and cancelled context.
    98  func (sch *Scheduler) Run(ctx context.Context) RunStatus {
    99  	// create sleeping workers waiting data on channel
   100  	for w := 1; w <= sch.Config().Resource.CronWorkers; w++ {
   101  		workerCtx, cancel := context.WithCancel(ctx)
   102  		defer cancel()
   103  		go sch.chainWorker(workerCtx, sch.chainsChan)
   104  	}
   105  	for w := 1; w <= sch.Config().Resource.IntervalWorkers; w++ {
   106  		workerCtx, cancel := context.WithCancel(ctx)
   107  		defer cancel()
   108  		go sch.intervalChainWorker(workerCtx, sch.ichainsChan)
   109  	}
   110  	ctx = log.WithLogger(ctx, sch.l)
   111  
   112  	/*
   113  		Loop forever or until we ask it to stop.
   114  		First loop fetches notifications.
   115  		Main loop works every refetchTimeout seconds and runs chains.
   116  	*/
   117  	sch.l.Info("Accepting asynchronous chains execution requests...")
   118  	go sch.retrieveAsyncChainsAndRun(ctx)
   119  
   120  	if sch.Config().Start.Debug { //run blocking notifications receiving
   121  		sch.pgengine.HandleNotifications(ctx)
   122  		return ContextCancelledStatus
   123  	}
   124  
   125  	sch.l.Debug("Checking for @reboot task chains...")
   126  	sch.retrieveChainsAndRun(ctx, true)
   127  
   128  	// Use ticker for strict intervals
   129  	ticker := time.NewTicker(refetchTimeout * time.Second)
   130  	defer ticker.Stop()
   131  	for {
   132  		sch.l.Debug("Checking for task chains...")
   133  		go sch.retrieveChainsAndRun(ctx, false)
   134  		sch.l.Debug("Checking for interval task chains...")
   135  		go sch.retrieveIntervalChainsAndRun(ctx)
   136  
   137  		select {
   138  		case <-ticker.C:
   139  			// pass
   140  		case <-ctx.Done():
   141  			sch.status = ContextCancelledStatus
   142  		case <-sch.shutdown:
   143  			sch.status = ShutdownStatus
   144  			sch.terminateChains()
   145  		}
   146  		if sch.status != RunningStatus {
   147  			return sch.status
   148  		}
   149  	}
   150  }
